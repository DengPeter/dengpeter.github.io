pts

打开terminal4个窗口分别登陆到同一台远程服务器,此时查看

$ ls /dev/pts/  #查看登陆的终端编号
1  2  3  4  ptmx

$ ps #查当前自己的终端编号
  PID TTY          TIME CMD
  23446 pts/2    00:00:00 bash
  23879 pts/2    00:00:00 ps

$ ps aux|grep pts #查看所有登陆的终端
test     23408  0.0  0.2 133672  2204 ?        S    07:11   0:00 sshd: test@pts/1
test     23409  0.0  0.3 116752  3492 pts/1    Ss   07:11   0:00 -bash
test     23445  0.0  0.2 133672  2076 ?        S    07:11   0:00 sshd: test@pts/2
test     23446  0.0  0.3 116672  3464 pts/2    Ss   07:11   0:00 -bash
test     23482  0.0  0.2 133672  2084 ?        S    07:12   0:00 sshd: test@pts/3
test     23483  0.0  0.3 116540  3244 pts/3    Ss+  07:12   0:00 -bash
test     23522  0.0  0.2 133672  2080 ?        S    07:12   0:00 sshd: test@pts/4
test     23523  0.0  0.3 116540  3228 pts/4    Ss+  07:12   0:00 -bash
test     23593  0.0  0.2 191320  2732 pts/1    S    07:14   0:00 sudo bash
test     23594  0.0  0.3 116728  3456 pts/1    S+   07:14   0:00 bash
test     23918  0.0  0.1 123356  1380 pts/2    R+   07:28   0:00 ps aux
test     23919  0.0  0.0 112656   964 pts/2    S+   07:28   0:00 grep --color=auto pts

$ echo 你好 |sudo tee /dev/pts/3 #向其它终端中发送信息，此时你会在打开的其它窗口看到echo出来的信息
