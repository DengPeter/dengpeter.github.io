python操作mongodb 

  import pymongo
  import time
  # 数据库连接
  client = pymongo.MongoClient('localhost',27017) # 连接到数据库
  db = client.test # 连接test库
  db.authenticate('user','password') #用户认证
  collection = db.test #连接test集合,相当于mysql的表

  # 数据操作
  data = {
    "id":1,
    "title":"python mongodb test document",
    "menu": ["menu1","menu2","menu3"],
    "body": {"menu1":"body1","menu2":'body2',"meun3":'body3'},
    "date": time.strftime('%Y-%m-%d %H:%M:S'),
  }

  ＃ 插入数据
  collection.insert(data) # 在集合中插入数据

  # 批量插入数据
  new_data = [
    {
     "id":2,
     "title":"python mongodb test document",
     "menu": ["menu1","menu2","menu3"],
     "body": {"menu1":"body1","menu2":'body2',"meun3":'body3'},
     "date": time.strftime('%Y-%m-%d %H:%M:S'),
   },
   {
    "id":3,
    "title":"python mongodb test document",
    "menu": ["menu1","menu2","menu3"],
    "body": {"menu1":"body1","menu2":'body2',"meun3":'body3'},
    "date": time.strftime('%Y-%m-%d %H:%M:S'),
  }
  ]

  # 删除数据
  collection.remove() # 删除集合中所有的内容
  collection.remove({'id':1}) # 删除集合中所有id为1的内容，一般id都是唯一的
