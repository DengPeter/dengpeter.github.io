基本选择器

console.log($("#content")); // ID选择器
console.log($('p'));        // 标签选择器
console.log($('.left'));    // 类选择器<div class='left'></div>
console.log($('*'));        // 选择所有元素
console.log($('div,p,.content')); //组合选择器

console.log($(':empty'))

层次选择器

console.log($('div span')); //选择div中 span的后代元素
console.log($('div > span')); //选择div中 span的子元素
console.log($('.one + div')); //选择one的下一个div元素 相当于：$(.one).next('div')
console.log($('#two ~ div ')); // 选择id为two的后面所有div兄弟元素 相当于：$('#two').nextAll('div')

过滤选择器

console.log($('div:fist')); //所有div中的第一个元素
console.log($('div:last')); //所有div中的最后一个元素
console.log($('input:not(.input-text')); //class名字不是input-text 的所有input标签
console.log($('li:even')); //索引数为偶数的li标签
console.log($('li:odd')); //索引数为奇数的li标签
console.log($('li:eq(1)')); //索引数为1的li标签 index以0开始
console.log($('li:gt(1')); //索引大于1
console.log($('li:lt'));   //索引小于1
console.log($(':header'));  // h系列标签选择器
console.log($(':empty')); //所有空元素
console.log($('div:empty')); //所有空div元素
console.log($('p:contains('内容')); //含有文本'内容' 的p标签。
console.log($('div:parent')); //包含子元素的div标签
console.log($('div:[id]')); //含有id属性的div标签
console.log($(':hidden')); //所有不可见元素
console.log($('div:hidden')); //所有不可见的div元素
console.log($('input[name=choose]')); //input中name是choose的标签

元素操作

### 添加：
    append() // 在被选元素的结尾插入内容
    prepend() // 在被选元素的开头插入内容
    after()  // 在被选元素之后插入内容
    before() // 在被选元素之前插入内容

### 删除：
    remove() // 删除被选元素（及其子元素)
    empty() // 从被选元素中删除子元素

### 复制：
    $(this).clone('ul').appendTo('div')  //复制当前节点中的 ul标签及子元素 添加到当前div中

### 替换
    $("p").replaceWith("<strong>this is a replace example</Strong>");

### 获取子元素个数

    $(this).child('p').length

checkbox操作

//判断checkbox是否被选中
if($('input[name=choose')[0].checked){
    console.log('选中')
}else{
    console.log('未选中')
}

//改变选中状态（反选)

 if($('input[name=choose')[0].checked){
    console.log('选中')
    $('input[name=choose')[0].checked == false;
}else{
    console.log('未选中')
    $('input[name=choose')[0].checked == true;
}
